数据同步机制
系统使用 sync_status 字段管理同步状态：
"local_new": 本地新建，待上传
"local_modified": 本地修改，待上传
"synced": 已同步
"deleted": 已删除   
通过 last_modified 时间戳判断数据版本，确保数据一致性。

系统采用 "客户端优先，云端同步" 的存储策略：用户操作 → 客户端存储 → 标记同步状态 → 云端同步

// 生成固定章节（按原始顺序）
function generateChapters(reviewWords: IWordReviewRecord[], date: string): Chapter[] {
  // 按原始顺序排列，确保章节内容顺序稳定

  const orderedWords = [...reviewWords]; // 保持原始顺序
  
  const chapters: Chapter[] = [];
  const CHAPTER_SIZE = 20; // 固定20个单词
  
  for (let i = 0; i < orderedWords.length; i += CHAPTER_SIZE) {
    const chapterWords = orderedWords.slice(i, i + CHAPTER_SIZE);
    const chapterNumber = Math.floor(i / CHAPTER_SIZE) + 1;
    
    const completedWords = chapterWords.filter(w => w.todayPracticeCount > 0).length;
    const isCompleted = chapterWords.every(w => w.todayPracticeCount > 0);
    
    chapters.push({
      chapterNumber,
      words: chapterWords,
      totalWords: chapterWords.length,
      completedWords,
      isCompleted,
      practiceCount: 0 // 从localStorage获取
    });
  }
  
  return chapters;
}

5. 本地存储管理

// src/utils/chapterStorage.ts
const CHAPTER_PRACTICE_COUNTS_KEY = 'chapter_practice_counts';

export function getChapterPracticeCounts(date: string): Record<number, number> {
  try {
    const key = `${CHAPTER_PRACTICE_COUNTS_KEY}_${date}`;
    const stored = localStorage.getItem(key);
    return stored ? JSON.parse(stored) : {};
  } catch (error) {
    console.error('获取章节练习次数失败:', error);
    return {};
  }
}

export function updateChapterPracticeCount(date: string, chapterNumber: number) {
  try {
    const key = `${CHAPTER_PRACTICE_COUNTS_KEY}_${date}`;
    const counts = getChapterPracticeCounts(date);
    counts[chapterNumber] = (counts[chapterNumber] || 0) + 1;
    localStorage.setItem(key, JSON.stringify(counts));
  } catch (error) {
    console.error('更新章节练习次数失败:', error);
  }
}

// 清理过期的章节数据
export function cleanupExpiredChapterData() {
  try {
    const today = new Date().toISOString().split('T')[0];
    const keys = Object.keys(localStorage);
    
    keys.forEach(key => {
      if (key.startsWith(CHAPTER_PRACTICE_COUNTS_KEY)) {
        const date = key.replace(CHAPTER_PRACTICE_COUNTS_KEY + '_', '');
        if (date !== today) {
          localStorage.removeItem(key);
        }
      }
    });
  } catch (error) {
    console.error('清理过期章节数据失败:', error);
  }
}
